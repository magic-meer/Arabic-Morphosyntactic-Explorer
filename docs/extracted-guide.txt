      - uses: actions/checkout@v2
      - name: Set up Python

      - name: Install dependencies
        run: pip install -r requirements.txt
      - name: Deploy to Azure
        uses: azure/webapps-deploy@v2
        with:

          publish-profile: ${{ secrets.AZURE_PUBLISH_PROFILE }}
          package: .
